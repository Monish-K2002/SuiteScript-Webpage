This function is used to retrieve the search data. 

const getSearchData = (searchId, filters) => {
    let searchData = search.load(searchId);
    let finalResult = [];
    if(filters.length>0){
        let defaultFilters = searchData.filters;
        defaultFilters.push(...filters);
        searchData.filters = defaultFilters;
    }
    let count = searchData.runPaged().count;
    let start = 0;
    let end = 1000;
    do{
        let result = searchData.run().getRange(start,end);
        result.forEach(data => finalResult.push(data));
        start +=1000;
        end += 1000;
    }
    while(start < count);
    return finalResult;
};
This function is used to check the null values mostly used along with CSV and Excel Values. 

const checkNull = (value) => {
    if(value == '' || value == ' ' || value == null || value == undefined){
        return '';
    }
    else{
        return value;
    }
